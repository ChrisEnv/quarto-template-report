% definiere Funktion: positioniert ein Bild an einer bestimmten Stelle. argumente: x-pos, y-pos, optionen, pfad zum bild
\def\placefig#1#2#3#4#5{
  \begin{textblock}{#3}(#1,#2)
    \rlap{\includegraphics[#4]{#5}}
  \end{textblock}}
